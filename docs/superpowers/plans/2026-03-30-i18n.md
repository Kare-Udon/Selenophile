# Selenophile i18n Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 Selenophile 增加跟随系统语言的界面本地化能力，并允许用户在设置页手动切换界面语言。

**Architecture:** 在 `SelenophileKit` 中新增一个轻量语言模型和本地化访问层，负责把用户偏好与系统语言解析成当前生效语言。应用层通过可观察的语言状态对象向 SwiftUI 注入 `Locale`，并在 AppKit 窗口标题等非 SwiftUI 文本处显式读取当前语言对应字符串。界面文案统一迁移到 `Localizable.strings`，设置页新增语言选择并走现有配置持久化链路。

**Tech Stack:** Swift 6.2, SwiftUI, AppKit, Foundation Localization, Swift Testing, Swift Package Manager, Xcode/Tuist project resources

```swift
import Foundation

public enum AppLanguage: String, CaseIterable, Codable, Sendable {
    case system
    case english = "en"
    case simplifiedChinese = "zh-Hans"
    case spanish = "es"
    case hindi = "hi"
    case arabic = "ar"
    case brazilianPortuguese = "pt-BR"
    case bengali = "bn"
    case russian = "ru"
    case japanese = "ja"
    case punjabi = "pa"
    case german = "de"
    case javanese = "jv"
    case korean = "ko"
    case french = "fr"
    case telugu = "te"
    case marathi = "mr"
    case turkish = "tr"
    case tamil = "ta"
    case vietnamese = "vi"
    case urdu = "ur"

    public static let supportedSelections: [AppLanguage] = [
        .system, .english, .simplifiedChinese, .spanish, .hindi, .arabic,
        .brazilianPortuguese, .bengali, .russian, .japanese, .punjabi,
        .german, .javanese, .korean, .french, .telugu, .marathi,
        .turkish, .tamil, .vietnamese, .urdu
    ]

    public var localeIdentifier: String {
        switch self {
        case .system:
            return Self.english.rawValue
        default:
            return rawValue
        }
    }

    public func resolved(preferredLanguages: [String] = Locale.preferredLanguages) -> AppLanguage {
        self == .system ? Self.resolvePreferredLanguage(from: preferredLanguages) : self
    }

    public static func resolvePreferredLanguage(from preferredLanguages: [String]) -> AppLanguage {
        for identifier in preferredLanguages {
            let normalized = identifier.lowercased()
            if normalized.hasPrefix("zh") { return .simplifiedChinese }
            if normalized.hasPrefix("en") { return .english }
            if normalized.hasPrefix("es") { return .spanish }
            if normalized.hasPrefix("hi") { return .hindi }
            if normalized.hasPrefix("ar") { return .arabic }
            if normalized.hasPrefix("pt") { return .brazilianPortuguese }
            if normalized.hasPrefix("bn") { return .bengali }
            if normalized.hasPrefix("ru") { return .russian }
            if normalized.hasPrefix("ja") { return .japanese }
            if normalized.hasPrefix("pa") { return .punjabi }
            if normalized.hasPrefix("de") { return .german }
            if normalized.hasPrefix("jv") { return .javanese }
            if normalized.hasPrefix("ko") { return .korean }
            if normalized.hasPrefix("fr") { return .french }
            if normalized.hasPrefix("te") { return .telugu }
            if normalized.hasPrefix("mr") { return .marathi }
            if normalized.hasPrefix("tr") { return .turkish }
            if normalized.hasPrefix("ta") { return .tamil }
            if normalized.hasPrefix("vi") { return .vietnamese }
            if normalized.hasPrefix("ur") { return .urdu }
        }
        return .english
    }
}
```

- [ ] **Step 3: 写最小实现，把语言选择纳入配置模型和持久化链路**

```swift
public struct MoonrakerConfiguration: Codable, Equatable, Sendable {
    public var serverURLString: String
    public var apiToken: String?
    public var cameraSnapshotURL: String?
    public var appLanguage: AppLanguage

    public init(
        serverURLString: String,
        apiToken: String?,
        cameraSnapshotURL: String? = nil,
        appLanguage: AppLanguage = .system
    ) {
        self.serverURLString = serverURLString.trimmingCharacters(in: .whitespacesAndNewlines)
        self.apiToken = apiToken?.trimmingCharacters(in: .whitespacesAndNewlines).flatMap { $0.isEmpty ? nil : $0 }
        self.cameraSnapshotURL = cameraSnapshotURL?.trimmingCharacters(in: .whitespacesAndNewlines).flatMap { $0.isEmpty ? nil : $0 }
        self.appLanguage = appLanguage
    }

    private enum CodingKeys: String, CodingKey {
        case serverURLString
        case apiToken
        case cameraSnapshotURL
        case appLanguage
    }

    public init(from decoder: any Decoder) throws {
        let container = try decoder.container(keyedBy: CodingKeys.self)
        serverURLString = try container.decode(String.self, forKey: .serverURLString)
        apiToken = try container.decodeIfPresent(String.self, forKey: .apiToken)
        cameraSnapshotURL = try container.decodeIfPresent(String.self, forKey: .cameraSnapshotURL)
        appLanguage = try container.decodeIfPresent(AppLanguage.self, forKey: .appLanguage) ?? .system
        self = .init(
            serverURLString: serverURLString,
            apiToken: apiToken,
            cameraSnapshotURL: cameraSnapshotURL,
            appLanguage: appLanguage
        )
    }
}
```

```swift
import Foundation

public struct AppLocalization: Sendable {
    public let language: AppLanguage

    public init(language: AppLanguage) {
        self.language = language.resolved()
    }

    public func string(_ key: String) -> String {
        let bundle = Bundle.module
        guard
            let path = bundle.path(forResource: language.localeIdentifier, ofType: "lproj"),
            let localizedBundle = Bundle(path: path)
        else {
            return NSLocalizedString(key, bundle: bundle, comment: "")
        }

        return NSLocalizedString(key, bundle: localizedBundle, comment: "")
    }
}
```
